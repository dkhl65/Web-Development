//Soft pot, PULL-UP. Current scale: C4 Major (I think)
// * note 	frequency 
// * c 	        261 Hz
// * d 	        294 Hz
// * e 	        329 Hz
// * f 	        349 Hz
// * g 	        392 Hz
// * a 	        440 Hz
// * b 	        493 Hz
// * C	        523 Hz
int notes[] = { 
  261, 294, 329, 349, 392, 440, 493, 523 };
char names[] = { 
  'c','d','e','f','g','a','b','C' };
const int pot = 0;
const int piezo = 9;
const int button = 2;
int potval;
void setup(){
  pinMode(button, INPUT);
  pinMode(piezo, OUTPUT);
  Serial.begin(9600);
}
void loop(){
  potval = analogRead(pot);
  potval = map(potval, 0, 1023, 0, 7);
  if (analogRead(pot) > 1018){
    potval = 7;
  }
  //"mary had a little lamb" setting
  //potval = map(potval, 0, 1023, 0, 5);
  Serial.print(names[potval]);
  Serial.print("\t");
  Serial.print(analogRead(pot));
  Serial.print("\n");
  if (digitalRead(button) == LOW){
    tone(piezo, notes[potval]);
  }
  else{
    noTone(piezo);
  }
  delay(75);
}
